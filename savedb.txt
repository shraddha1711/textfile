CREATE TABLE Users
(
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Username NVARCHAR(100) NOT NULL,
    Age INT NOT NULL,
    Gender NVARCHAR(10) NOT NULL,
    Country NVARCHAR(50) NOT NULL,
    ImagePath NVARCHAR(200),
    FinalSubmit BIT NOT NULL
);



public class User
{
    public int Id { get; set; }
    public string Username { get; set; }
    public int Age { get; set; }
    public string Gender { get; set; }
    public string Country { get; set; }
    public string ImagePath { get; set; }
    public bool FinalSubmit { get; set; }
}



using System.Data.Entity;

public class ApplicationDbContext : DbContext
{
    public ApplicationDbContext() : base("DefaultConnection") { }
    public DbSet<User> Users { get; set; }
}

using System.Linq;
using System.Web;
using System.Web.Mvc;
using YourNamespace.Models;

public class UserController : Controller
{
    private ApplicationDbContext db = new ApplicationDbContext();

    // GET: Index → show form + table
    public ActionResult Index()
    {
        var users = db.Users.OrderBy(u => u.Id).ToList();
        return View(users);
    }

    // POST: Save (Create or Update)
    [HttpPost]
    public ActionResult Save(User model, HttpPostedFileBase profileImage)
    {
        if(ModelState.IsValid)
        {
            if(profileImage != null)
            {
                var path = Server.MapPath("~/Images/");
                System.IO.Directory.CreateDirectory(path);
                var fileName = System.IO.Path.GetFileName(profileImage.FileName);
                profileImage.SaveAs(path + fileName);
                model.ImagePath = "/Images/" + fileName;
            }

            if(model.Id == 0)
                db.Users.Add(model);
            else
                db.Entry(model).State = System.Data.Entity.EntityState.Modified;

            db.SaveChanges();
            return RedirectToAction("Index");
        }
        return RedirectToAction("Index");
    }

    // GET: Delete
    public ActionResult Delete(int id)
    {
        var user = db.Users.Find(id);
        if(user != null)
        {
            db.Users.Remove(user);
            db.SaveChanges();
        }
        return RedirectToAction("Index");
    }

    // GET: Get user by id (for edit)
    public JsonResult GetUser(int id)
    {
        var user = db.Users.Find(id);
        return Json(user, JsonRequestBehavior.AllowGet);
    }
}

<script>
$(document).ready(function(){
    // Live validation
    $('#username').on('input', function(){
        const val = $(this).val();
        $('#usernameError').text(/^[A-Za-z ]*$/.test(val) ? "" : "Only letters allowed");
    });
    $('#age').on('input', function(){
        const val = $(this).val();
        $('#ageError').text(/^\d*$/.test(val) ? "" : "Only numbers allowed");
    });

    // Edit button click
    $('.editBtn').click(function(){
        const id = $(this).data('id');
        $.get('@Url.Action("GetUser","User")', { id: id }, function(data){
            $('input[name="Id"]').val(data.Id);
            $('input[name="Username"]').val(data.Username);
            $('input[name="Age"]').val(data.Age);
            $('input[name="Gender"][value="'+data.Gender+'"]').prop('checked', true);
            $('select[name="Country"]').val(data.Country);
            $('input[name="FinalSubmit"]').prop('checked', data.FinalSubmit);
        });
    });

    // Filter by Username
    $('#search').on('keyup', function(){
        const val = $(this).val().toLowerCase();
        $('#userTable tr').filter(function(){
            $(this).toggle($(this).find('td:eq(1)').text().toLowerCase().indexOf(val) > -1)
        });
    });
});
</script>
